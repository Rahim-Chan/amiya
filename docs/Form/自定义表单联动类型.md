# 自定义表单联动类型

> 注册过后，AySearchTable、AyDialogForm 也会生效

## 示例

1. 实现受控组件
2. 受控组件内部监听表单数据变化
3. 根据变化改变本身状态
4. 注册自定义表单类型
5. 使用组件

注册

```tsx
import React, { useState, useEffect } from 'react'
import { AyForm, AyButton, AySelect, registerField, Option, FieldListener } from 'amiya'
import 'antd/dist/antd.min.css'

interface JobSelectProps {
  value?: string
  onChange?: (value: string) => void
  placeholder?: string
  /** 添加 field 监听 */
  addFieldListener?: (key: string, fieldListener: FieldListener) => void
  /** 删除 field 监听 */
  removeFiledListener?: (key: string, fieldListener: FieldListener) => void
}

// 【模拟】延迟一秒获取职业列表
const getOptions = (value) => {
  return new Promise((resolve) => {
    setTimeout(() => {
      let options = []
      if (value === 'amiya') {
        // amiya 的可选职业
        options = [
          { label: '近卫干员', value: 'jinwei' },
          { label: '术师干员', value: 'shushi' }
        ]
      } else if (value === 'exusiai') {
        // exusiai 的可选职业
        options = [
          { label: '狙击干员', value: 'juji' },
          { label: '抛光干员', value: 'paoguang' }
        ]
      }
      resolve(options)
    }, 1000)
  })
}

// 职业选择
function JobSelect(props: any) {
  const { value, onChange, placeholder, addFieldListener, removeFiledListener } = props
  // 选项
  const [options, setOptions] = useState<Array<Option>>([])
  // 是否正在请求中
  const [loading, setLoading] = useState<boolean>(false)

  useEffect(() => {
    // 声明角色选择变化事件
    const handleCharaChange = (value: any) => {
      setLoading(true)
      getOptions(value)
        .then((options: any) => {
          setOptions(options)
        })
        .finally(() => {
          setLoading(false)
        })
      if (onChange) {
        onChange(null)
      }
    }
    // 添加监听事件
    addFieldListener && addFieldListener('chara', handleCharaChange)
    // 记得销毁
    return () => {
      removeFiledListener && removeFiledListener('chara', handleCharaChange)
    }
  }, [])

  return <AySelect loading={loading} value={value} onChange={onChange} options={options} placeholder={placeholder} />
}

// 注册职业选择
registerField('job-select', {
  type: 'job-select',
  defaultValue: null,
  render: ({ field, setFieldsValue, readonly, addFieldListener, removeFiledListener }: any) => {
    return (
      <JobSelect
        addFieldListener={addFieldListener}
        removeFiledListener={removeFiledListener}
        placeholder={`请选择${field.title || ''}`}
        {...field.props}
      />
    )
  }
})

const fields: Array<AyFormField> = [
  {
    title: '角色',
    key: 'chara',
    type: 'select',
    options: [
      {
        label: '阿米娅',
        value: 'amiya'
      },
      {
        label: '能天使',
        value: 'exusiai'
      }
    ]
  },
  {
    title: '职业选择',
    type: 'job-select',
    key: 'job'
  },
  // 注册过后，可以同时存在多个
  {
    title: '职业选择2',
    type: 'job-select',
    key: 'job2'
  }
]

export default function Demo() {
  const handleConfirm = (form: any) => {
    console.log(form)
    alert(JSON.stringify(form))
  }

  return (
    <AyForm span={24} fields={fields} style={{ width: 400, margin: '0 auto' }} onConfirm={handleConfirm}>
      <AyButton type="primary" htmlType="submit" style={{ marginLeft: 120 }}>
        提交
      </AyButton>
    </AyForm>
  )
}
```
