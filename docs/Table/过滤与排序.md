# 过滤与排序

```tsx
import React from 'react'
import { AySearchTable, AySearchTableField } from 'amiya'
import { professionOptions, listApi } from '../api'
import 'antd/dist/antd.min.css'

const fields: Array<AySearchTableField> = [
  {
    title: '姓名',
    key: 'cname',
    table: {
      width: 300
    }
  },
  {
    title: '初始HP',
    key: 'defaultHp',
    table: {
      sort: true,
      sortOrder: 1 // 排序顺序，数字越高权重越大
    }
  },
  {
    title: '初始攻击',
    key: 'defaultAtk',
    table: {
      sort: true,
      sortOrder: 2 // 排序顺序，数字越高权重越大
    }
  },
  {
    title: '星级(单选)',
    key: 'rarity',
    options: [
      { label: '⭐️', value: 0 },
      { label: '⭐️⭐️', value: 1 },
      { label: '⭐️⭐️⭐️', value: 2 },
      { label: '⭐️⭐️⭐️⭐️', value: 3 },
      { label: '⭐️⭐️⭐️⭐️⭐️', value: 4 },
      { label: '⭐️⭐️⭐️⭐️⭐️⭐️', value: 5 }
    ],
    table: {
      width: 300,
      sort: true,
      sortOrder: 3, // 排序顺序，数字越高权重越大
      filter: true
    }
  },
  {
    title: '职业(多选)',
    key: 'profession',
    type: 'select',
    options: professionOptions,
    table: {
      filter: true,
      filterMultiple: true
    }
  }
]

export default function Demo() {
  return <AySearchTable title="表格标题" searchVisible={false} api={listApi} fields={fields} />
}
```
