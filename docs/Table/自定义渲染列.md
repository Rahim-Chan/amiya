# 自定义渲染列

## 写 render 函数

可以通过 table 的 render 函数来自定义渲染这一列长什么样子。

```tsx
import React from 'react'
import { AySearchTable, AySearchTableField } from 'amiya'
import { Tag } from 'antd'
import 'antd/dist/antd.min.css'

const data = [
  {
    id: '1',
    name: 'Amiya',
    cname: '阿米娅',
    defaultHp: 699,
    defaultAtk: 276,
    job: '3',
    createDate: '2019年4月30日 10:00'
  },
  {
    id: '2',
    name: 'Exusiai',
    cname: '能天使',
    defaultHp: 711,
    defaultAtk: 183,
    job: '2',
    createDate: '2019年4月30日 10:00'
  }
]

const fields: Array<AySearchTableField> = [
  {
    title: '姓名',
    key: 'cname'
  },
  {
    title: '英文名',
    key: 'name',
    search: {},
    table: {
      render: (text: string, record: any, index: number) => {
        const colorMap: any = {
          Amiya: 'cyan',
          Exusiai: 'orange'
        }
        return <Tag color={colorMap[text]}>{text}</Tag>
      }
    }
  },
  {
    title: '初始HP',
    key: 'defaultHp'
  },
  {
    title: '初始攻击',
    key: 'defaultAtk'
  },
  {
    title: '职业',
    key: 'job',
    type: 'select',
    search: {},
    options: [
      { label: '狙击干员', value: '1' },
      { label: '医疗干员', value: '2' },
      { label: '术师干员', value: '3' }
    ]
  },
  {
    title: '上线时间',
    key: 'createDate'
  }
]

export default function Demo() {
  return <AySearchTable title="使用 render 函数" data={data} fields={fields} />
}
```

## 全局注册

如果每次都需要定义比较麻烦的话，可以尝试注册全局的 render，下面这个例子跟上一个 demo 效果一致。

```tsx
import React from 'react'
import { AySearchTable, AySearchTableField, registerTableRender, RenderProps } from 'amiya'
import { Tag } from 'antd'
import 'antd/dist/antd.min.css'

const data = [
  {
    id: '1',
    name: 'Amiya',
    cname: '阿米娅',
    defaultHp: 699,
    defaultAtk: 276,
    job: '3',
    createDate: '2019年4月30日 10:00'
  },
  {
    id: '2',
    name: 'Exusiai',
    cname: '能天使',
    defaultHp: 711,
    defaultAtk: 183,
    job: '2',
    createDate: '2019年4月30日 10:00'
  }
]

// 此函数只需要注册一次，你可以放在全局
registerTableRender('tag', ({ text, record, field }: RenderProps) => {
  return <Tag color={field.colorMap[text]}>{text}</Tag>
})

const fields: Array<AySearchTableField> = [
  {
    title: '姓名',
    key: 'cname'
  },
  {
    title: '英文名',
    key: 'name',
    search: {},
    table: {
      colorMap: {
        Amiya: 'cyan',
        Exusiai: 'orange'
      },
      renderType: 'tag'
    }
  },
  {
    title: '初始HP',
    key: 'defaultHp'
  },
  {
    title: '初始攻击',
    key: 'defaultAtk'
  },
  {
    title: '职业',
    key: 'job',
    type: 'select',
    search: {},
    options: [
      { label: '狙击干员', value: '1' },
      { label: '医疗干员', value: '2' },
      { label: '术师干员', value: '3' }
    ]
  },
  {
    title: '上线时间',
    key: 'createDate'
  }
]

export default function Demo() {
  return <AySearchTable title="全局注册渲染" data={data} fields={fields} />
}
```

```tsx
import { Tag, Popover, Badge, Progress, Space } from 'antd'

import { ClockCircleOutlined, InfoCircleOutlined } from '@ant-design/icons'

const data = [

    name: 'Amiya',
    cname: '阿米娅',
    defaultHp: 699,
    defaultAtk: 276,
    job: '3',
    status: '1',
    progress: 70,

    link: 'http://prts.wiki/w/%E9%98%BF%E7%B1%B3%E5%A8%85',
    info: `
【代号】阿米娅
【性别】女

    `
  },
  {
    id: '2',
    name: 'Exusiai',
    cname: '能天使',
    defaultHp: 711,
    defaultAtk: 183,
    job: '1',
    status: '0',
    createDate: '2019年4月30日 10:00',
    link: 'http://prts.wiki/w/%E8%83%BD%E5%A4%A9%E4%BD%BF',
    info: `
【代号】能天使
【性别】女
【战斗经验】两年
【出身地】拉特兰
【生日】12月24日
【种族】萨科塔
【身高】159cm
【矿石病感染情况】
参照医学检测报告，确认为非感染者。
    `
  }
]

// 此函数只需要注册一次，你可以放在全局
registerTableRender('tag', ({ text, record, field }: RenderProps) => {
  return <Tag color={field.colorMap[text]}>{text}</Tag>
})

// 此函数只需要注册一次，你可以放在全局
registerTableRender('job', ({ field, text }: RenderProps) => {
  let option = field.options.find((option: Option) => option.value === text)
  if (!option) {
    return text
  }
  return (
    <Space>
      <img src={option.url} />
      {option.label}
    </Space>
  )
})

// 此函数只需要注册一次，你可以放在全局
registerTableRender('datetime', ({ text, record, field }: RenderProps) => {
  return (
    <Space>
      <ClockCircleOutlined />
      <Tag color="purple">{text}</Tag>
    </Space>
  )
})

const statusMap = {
  0: {
    label: '待命中',
    status: 'success'
  },
  1: {
    label: '体力恢复中',
    color: 'blue',
    status: 'processing'
  }
}
// 此函数只需要注册一次，你可以放在全局
registerTableRender('status', ({ text, record, field }: RenderProps) => {
  const statusOption = statusMap[text]
  return (
    <>
      <Badge status={statusOption.status} color={statusOption.color} />
      {statusOption.label}
      {record.progress && <Progress status="active" percent={record.progress} />}
    </>
  )
})

// 此函数只需要注册一次，你可以放在全局
registerTableRender('cname', ({ text, record, field }: RenderProps) => {
  return (
    <Space>
      <span>{text}</span>
      <Popover
        content={
          <>
            <pre>{record.info ? record.info.trim() : '-'}</pre>
            {record.link && (
              <a href={record.link} target="_blank">
                查看百科
              </a>
            )}
          </>
        }
        title="干员信息"
      >
        <InfoCircleOutlined style={{ color: 'purple' }} />
      </Popover>
    </Space>
  )
})

const fields: Array<AySearchTableField> = [
  {
    title: '姓名',
    key: 'cname',
    table: {
      renderType: 'cname'
    }
  },
  {
    title: '英文名',
    key: 'name',
    search: {},
    table: {
      colorMap: {
        Amiya: 'cyan',
        Exusiai: 'orange'
      },
      renderType: 'tag'
    }
  },
  {
    title: '职业',
    key: 'job',
    type: 'select',
    search: {},
    table: {
      renderType: 'job'
    },
    options: [
      { label: '狙击干员', value: '1', url: require('./images/图标_职业_狙击.png') },
      { label: '医疗干员', value: '2', url: require('./images/图标_职业_医疗.png') },
      { label: '术师干员', value: '3', url: require('./images/图标_职业_术师.png') }
    ]
  },
  {
    title: '状态',
    key: 'status',
    table: {
      renderType: 'status'
    }
  },
  {
    title: '上线时间',
    key: 'createDate',
    table: {
      renderType: 'datetime'
    }
  }
]

export default function Demo() {
  return <AySearchTable title="全局注册渲染" data={data} fields={fields} />
}
```
